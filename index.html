bool checkBrackets(string& str, char** bracketsConfig, int size_of_braketsConfig){

stack<char>* stk = new stack<char>();
char brace;

for (int i = 0; i < str.size(); i++) {

if (stk->size() != 0) {

brace = find_second_brace(bracketsConfig, size_of_braketsConfig, str[i]);

if (stk->top() == brace) {
stk->pop();
continue;
}
else {
return false;
}
}

brace = find_first_brace(bracketsConfig, size_of_braketsConfig, str[i]);

if (brace == -1) {
return false;
}

stk->push(str[i]);
}

if (stk->size() == 0) {
delete stk;
return true;
}
else {
delete stk;
return false;
}
}


/*Function searching the opening brace in bracketsConfig*/
char find_first_brace(char** arr, int size, char brace) {

for (int i = 0; i < size; i++) {
if (arr[i][0] == brace) {
return arr[i][0];
}
}

return 0;
}


/*Function searching the closing brace in bracketsConfig*/
char find_second_brace(char** arr, int size, char brace) {

for (int i = 0; i < size; i++) {
if (arr[i][1] == brace) {
return arr[i][0];
}
}

return 0;
}
